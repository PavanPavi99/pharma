package com.pharmalens.serviceimpl; import java.util.List; import org.springframework.beans.factory.annotation.Autowired; import org.springframework.stereotype.Service; import com.pharmalens.entity.Medicine; import com.pharmalens.repository.BillRepository; import com.pharmalens.repository.CustomerRepository; import com.pharmalens.repository.MedicineRepository; import com.pharmalens.repository.SupplierRepository; import com.pharmalens.service.ReportService; @Service public class ReportServiceImpl implements ReportService { @Autowired private MedicineRepository medicineRepository; @Autowired private CustomerRepository customerRepository; @Autowired private SupplierRepository supplierRepository; @Autowired private BillRepository billRepository; @Override public long getTotalMedicines() { return medicineRepository.count(); } @Override public long getTotalCustomers() { return customerRepository.count(); } @Override public long getTotalSuppliers() { return supplierRepository.count(); } @Override public long getTotalBills() { return billRepository.count(); } @Override public double getTotalSales() { return billRepository.findAll() .stream() .mapToDouble(bill -> bill.getTotalAmount()) .sum(); } // Low Stock Medicine Report @Override public List getLowStockMedicines() { return medicineRepository.findLowStockMedicines(); } }
